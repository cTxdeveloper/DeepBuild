```python
import threading
from src.task_queue import TaskQueue
from src.worker import WorkerThread

class AgentCLI:
    def __init__(self, queue: TaskQueue):
        self.queue = queue
        self.worker = WorkerThread(queue)

    def start(self):
        print("🤖 Agent OS ready. Type /goal <description> to start, /status for progress, /exit to stop.")
        self.worker.start()
        try:
            while True:
                user_input = input("> ").strip()
                if not user_input:
                    continue
                parts = user_input.split(maxsplit=1)
                cmd = parts[0].lower()
                args = parts[1] if len(parts) > 1 else ""
                if cmd == "/goal":
                    self._cmd_goal(args)
                elif cmd == "/status":
                    self._cmd_status()
                elif cmd == "/exit":
                    self._cmd_exit()
                    break
                elif cmd == "/help":
                    print("Commands: /goal <desc>, /status, /exit")
                else:
                    print(f"Unknown command: {cmd}")
        except KeyboardInterrupt:
            print("\nInterrupted. Shutting down...")
        finally:
            self.worker.stop()
            self.worker.join()
            print("Agent stopped.")

    def _cmd_goal(self, description):
        if not description:
            print("Usage: /goal <project description>")
            return
        goal_id = self.queue.add_goal(description)
        # Immediately add a planning task for the new goal
        self.queue.add_task(goal_id, f"Plan the entire project: {description}", "orchestrator")
        print(f"✅ Goal #{goal_id} created and planning started: {description}")

    def _cmd_status(self):
        goals = self.queue.get_all_goals()
        if not goals:
            print("No goals yet.")
            return
        for g in goals:
            gid, desc, pri, status, _ = g
            pending = self.queue.count_tasks(gid, "pending")
            in_progress = self.queue.count_tasks(gid, "in_progress")
            completed = self.queue.count_tasks(gid, "completed")
            failed = self.queue.count_tasks(gid, "failed")
            print(f"Goal #{gid} ({status}) – {desc[:60]}...")
            print(f"   Tasks: Pending {pending}, InProgress {in_progress}, Completed {completed}, Failed {failed}")

    def _cmd_exit(self):
        print("Shutting down worker...")
        self.worker.stop()
```

src/worker.py

```python
import time
import threading
from src.task_queue import TaskQueue
from src.crew import KickoffCrew
from src.memory.manager import memory_manager  # global instance

class WorkerThread(threading.Thread):
    def __init__(self, queue: TaskQueue):
        super().__init__(daemon=True)
        self.queue = queue
        self._stop_event = threading.Event()

    def run(self):
        print("⚙️  Worker loop started.")
        while not self._stop_event.is_set():
            task = self.queue.get_next_pending_task()
            if not task:
                time.sleep(5)
                continue
            task_id, goal_id, description, agent_role, status, output = task
            print(f"\n🔨 Task #{task_id} [{agent_role}]: {description[:60]}...")
            self.queue.update_task_status(task_id, "in_progress")
            memory_manager.add_message("system", f"Starting task: {description}")

            try:
                inputs = {
                    "user_request": description,
                    "story_description": description,
                    "goal_id": goal_id
                }
                crew_instance = KickoffCrew(self.queue)
                crew = crew_instance.crew()
                result = crew.kickoff(inputs=inputs)
                output_str = str(result)
                self.queue.update_task_status(task_id, "completed", output_str)
                memory_manager.add_message("system", f"Task #{task_id} completed.")

                # Check if all tasks for this goal are done
                if self.queue.count_tasks(goal_id, "pending") == 0 and self.queue.count_tasks(goal_id, "in_progress") == 0:
                    self.queue.update_goal_status(goal_id, "completed")
                    print(f"🎉 Goal #{goal_id} fully completed!")
            except Exception as e:
                print(f"❌ Task #{task_id} failed: {e}")
                self.queue.update_task_status(task_id, "failed", str(e))
                memory_manager.store_as_memory(f"error_task_{task_id}", str(e))

    def stop(self):
        self._stop_event.set()
```

```python
import litellm
from litellm import completion
from typing import List, Dict

class TieredRouter:
    COMPLEXITY_THRESHOLD = 0.7
    SECURITY_KEYWORDS = ["auth", "payment", "encrypt", "private key", "password"]

    ROUTES = {
        "complex": "deepseek-pro",
        "bulk": "deepseek-flash",
        "research": "llama-3.3-70b",
        "vision": "gemini-2.5-flash",
        "code_review": "deepseek-pro"
    }

    def assess_complexity(self, text: str) -> float:
        length_factor = min(len(text) / 500, 1.0)
        security_penalty = 0.3 if any(kw in text.lower() for kw in self.SECURITY_KEYWORDS) else 0
        return min(length_factor + security_penalty, 1.0)

    def route_task(self, task_description: str, task_type: str = "code") -> str:
        if task_type == "vision":
            return self.ROUTES["vision"]
        if task_type == "research":
            return self.ROUTES["research"]
        complexity = self.assess_complexity(task_description)
        if complexity > self.COMPLEXITY_THRESHOLD or any(kw in task_description.lower() for kw in self.SECURITY_KEYWORDS):
            return self.ROUTES["complex"]
        return self.ROUTES["bulk"]

    def run_completion(self, messages: List[Dict], model_override: str = None, **kwargs):
        model = model_override or self.ROUTES["bulk"]
        response = completion(model=model, messages=messages, **kwargs)
        return response

router = TieredRouter()

class RouterLLM:
    """Wrapper to be used as a CrewAI LLM, dynamically selects the cheapest capable model."""
    def __init__(self, default_model="deepseek-flash"):
        self.default_model = default_model

    def call(self, messages, **kwargs):
        msg = messages[0]["content"].lower()
        if "search" in msg or "research" in msg:
            model = "llama-3.3-70b"
        elif "screenshot" in msg or "analyze this image" in msg:
            model = "gemini-2.5-flash"
        else:
            model = router.route_task(messages[0]["content"])
        return router.run_completion(messages, model_override=model, **kwargs)
```

```python
import sqlite3
from datetime import datetime

class TaskQueue:
    def __init__(self, db_path="tasks.db"):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self._init_db()

    def _init_db(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS goals (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                description TEXT,
                priority INTEGER DEFAULT 1,
                status TEXT DEFAULT 'pending',
                created_at TEXT
            )
        """)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS tasks (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                goal_id INTEGER,
                description TEXT,
                agent_role TEXT DEFAULT 'coder',
                status TEXT DEFAULT 'pending',
                output TEXT,
                created_at TEXT,
                FOREIGN KEY(goal_id) REFERENCES goals(id)
            )
        """)
        self.conn.commit()

    def add_goal(self, description, priority=1):
        now = datetime.now().isoformat()
        self.conn.execute(
            "INSERT INTO goals (description, priority, status, created_at) VALUES (?, ?, 'pending', ?)",
            (description, priority, now)
        )
        self.conn.commit()
        return self.conn.execute("SELECT last_insert_rowid()").fetchone()[0]

    def add_task(self, goal_id, description, agent_role="coder"):
        now = datetime.now().isoformat()
        self.conn.execute(
            "INSERT INTO tasks (goal_id, description, agent_role, status, created_at) VALUES (?, ?, ?, 'pending', ?)",
            (goal_id, description, agent_role, now)
        )
        self.conn.commit()

    def get_next_pending_task(self):
        cursor = self.conn.execute(
            "SELECT * FROM tasks WHERE status='pending' ORDER BY id ASC LIMIT 1"
        )
        return cursor.fetchone()

    def update_task_status(self, task_id, status, output=None):
        self.conn.execute(
            "UPDATE tasks SET status=?, output=? WHERE id=?",
            (status, output, task_id)
        )
        self.conn.commit()

    def update_goal_status(self, goal_id, status):
        self.conn.execute("UPDATE goals SET status=? WHERE id=?", (status, goal_id))
        self.conn.commit()

    def get_all_goals(self):
        return self.conn.execute("SELECT * FROM goals ORDER BY id DESC").fetchall()

    def count_tasks(self, goal_id, status):
        cursor = self.conn.execute(
            "SELECT COUNT(*) FROM tasks WHERE goal_id=? AND status=?", (goal_id, status)
        )
        return cursor.fetchone()[0]
```

```python
import chromadb
from chromadb.utils import embedding_functions
import litellm
from typing import List, Dict
import hashlib

class MemoryManager:
    def __init__(self, collection_name="agent_memory"):
        self.client = chromadb.PersistentClient(path=".memory")
        self.embed_fn = embedding_functions.DefaultEmbeddingFunction()
        self.collection = self.client.get_or_create_collection(
            name=collection_name, embedding_function=self.embed_fn
        )
        self.error_collection = self.client.get_or_create_collection(
            name="error_fixes", embedding_function=self.embed_fn
        )
        self.recent_messages: List[Dict] = []
        self.long_term_summary: str = ""
        self.max_recent = 12

    def add_message(self, role: str, content: str):
        self.recent_messages.append({"role": role, "content": content})
        if len(self.recent_messages) > self.max_recent:
            oldest = self.recent_messages.pop(0)
            self._update_summary(oldest["content"])

    def _update_summary(self, message: str):
        if not self.long_term_summary:
            self.long_term_summary = message
            return
        prompt = f"Previous summary: {self.long_term_summary}\nNew message: {message}\nMerge into a concise summary of the overall context. Keep it under 200 words."
        try:
            response = litellm.completion(
                model="deepseek-flash",
                messages=[{"role": "user", "content": prompt}],
                max_tokens=200
            )
            self.long_term_summary = response.choices[0].message.content
        except Exception:
            self.long_term_summary += "\n" + message

    def build_context(self, system_prompt: str, user_message: str, relevant_query: str = None) -> str:
        memories = self.retrieve_relevant(relevant_query or user_message, n_results=3)
        memory_text = "\n".join([f"- {m}" for m in memories]) if memories else ""
        context = system_prompt
        if memory_text:
            context += f"\n\n[RELEVANT PAST KNOWLEDGE]\n{memory_text}"
        if self.long_term_summary:
            context += f"\n\n[CONVERSATION SUMMARY]\n{self.long_term_summary}"
        recent = "\n".join([f"{m['role']}: {m['content']}" for m in self.recent_messages[-6:]])
        if recent:
            context += f"\n\n[RECENT CHAT]\n{recent}"
        context += f"\n\n[NEW MESSAGE]\n{user_message}"
        return context

    def store_as_memory(self, key: str, content: str, metadata: Dict = None):
        self.collection.add(documents=[content], metadatas=[metadata or {}], ids=[key])

    def retrieve_relevant(self, query: str, n_results=3) -> List[str]:
        results = self.collection.query(query_texts=[query], n_results=n_results)
        return results['documents'][0] if results['documents'] else []

    def store_error_fix(self, error_snippet: str, solution: str, metadata: dict = None):
        doc = f"ERROR: {error_snippet}\nSOLUTION: {solution}"
        metadata = metadata or {}
        metadata["type"] = "error_fix"
        uid = hashlib.md5(error_snippet.encode()).hexdigest()[:12]
        self.error_collection.add(documents=[doc], metadatas=[metadata], ids=[uid])

    def find_error_fix(self, error_message: str, n_results=2) -> List[str]:
        results = self.error_collection.query(query_texts=[error_message], n_results=n_results)
        if results['documents']:
            return results['documents'][0]
        return []

    def store_lesson(self, topic: str, insight: str):
        self.store_as_memory(key=f"lesson_{topic}", content=f"TOPIC: {topic}\nINSIGHT: {insight}")

# Global memory instance
memory_manager = MemoryManager()
```

```python
from src.memory.manager import MemoryManager

class RememberFactTool(BaseTool):

class RecallTool(BaseTool):

    description: str = "Search past knowledge and experiences. Input: { query: string }"

        super().__init__()
        self.memory = memory_manager
```

src/plugins/base.py

```python
import os, json, subprocess
from abc import ABC, abstractmethod
from typing import Dict

class BasePlugin(ABC):
    def __init__(self, plugin_path: str, manifest: Dict, config: Dict = None):
        self.plugin_path = plugin_path
        self.manifest = manifest
        self.config = config or {}
        self.instructions = self._load_instructions()
        self._install_dependencies()

    def _load_instructions(self) -> str:
        instructions_path = os.path.join(self.plugin_path, "instructions.md")
        if os.path.exists(instructions_path):
            with open(instructions_path, "r") as f:
                return f.read()
        folder = os.path.join(self.plugin_path, "instructions")
        if os.path.isdir(folder):
            combined = []
            for fname in sorted(os.listdir(folder)):
                if fname.endswith(".md"):
                    with open(os.path.join(folder, fname), "r") as f:
                        combined.append(f.read())
            return "\n\n".join(combined)
        return ""

    def _install_dependencies(self):
        deps = self.manifest.get("python_dependencies", [])
        if deps:
            subprocess.check_call(["pip", "install", *deps])

    @abstractmethod
    def run(self, **kwargs) -> str:
        pass

    def get_skill_prompt(self) -> str:
        return self.instructions
```

src/plugins/manager.py

```python
import os, json, importlib.util
from typing import Dict, List
from src.plugins.base import BasePlugin
from crewai_tools import BaseTool as CrewAIBaseTool

class PluginManager:
    def __init__(self, plugins_dir: str = "plugins"):
        self.plugins_dir = plugins_dir
        self.plugins: Dict[str, BasePlugin] = {}
        self.load_all()

    def load_all(self):
        if not os.path.isdir(self.plugins_dir):
            os.makedirs(self.plugins_dir, exist_ok=True)
            return
        for name in os.listdir(self.plugins_dir):
            plugin_path = os.path.join(self.plugins_dir, name)
            if not os.path.isdir(plugin_path):
                continue
            manifest_path = os.path.join(plugin_path, "manifest.json")
            if not os.path.exists(manifest_path):
                continue
            with open(manifest_path, "r") as f:
                manifest = json.load(f)
            config = {}
            for env_var in manifest.get("env_vars", []):
                config[env_var] = os.environ.get(env_var)
                if not config[env_var]:
                    raise ValueError(f"Missing env var {env_var} for plugin {name}")

            handler_path = os.path.join(plugin_path, "handler.py")
            if os.path.exists(handler_path):
                spec = importlib.util.spec_from_file_location(f"plugin_{name}", handler_path)
                module = importlib.util.module_from_spec(spec)
                spec.loader.exec_module(module)
                plugin_class = None
                for attr_name in dir(module):
                    obj = getattr(module, attr_name)
                    if isinstance(obj, type) and issubclass(obj, BasePlugin) and obj is not BasePlugin:
                        plugin_class = obj
                        break
                if plugin_class:
                    plugin_instance = plugin_class(plugin_path, manifest, config)
                    self.plugins[name] = plugin_instance
            else:
                # Pure skill – no handler
                class PureSkill(BasePlugin):
                    def run(self, **kwargs):
                        return "Skill activated."
                plugin_instance = PureSkill(plugin_path, manifest, {})
                self.plugins[name] = plugin_instance

    def get_tools_for_crew(self) -> List[CrewAIBaseTool]:
        tools = []
        for name, plugin in self.plugins.items():
            if plugin.manifest.get("type") == "tool":
                tools.append(self._create_crewai_tool(name, plugin))
        return tools

    def _create_crewai_tool(self, name: str, plugin: BasePlugin) -> CrewAIBaseTool:
        class PluginTool(CrewAIBaseTool):
            tool_name: str = name
            tool_description: str = plugin.manifest.get("description", "")
            def _run(self, **kwargs) -> str:
                return plugin.run(**kwargs)
        return PluginTool()

# Global plugin manager
plugin_manager = PluginManager()
```

```python
from youtube_transcript_api import YouTubeTranscriptApi
from crewai_tools import BaseTool
import re

class YouTubeTranscriptTool(BaseTool):
    name: str = "YouTube Transcript Extractor"
    description: str = "Extract subtitles from a YouTube video. Input: { video_url: string }"
    def _run(self, video_url: str) -> str:
        video_id = self._extract_id(video_url)
        if not video_id:
            return "Invalid YouTube URL."
        try:
            transcript = YouTubeTranscriptApi.get_transcript(video_id, languages=['en'])
            text = " ".join([entry['text'] for entry in transcript])
            return text[:5000]
        except Exception as e:
            return f"Transcript error: {e}"

    def _extract_id(self, url: str) -> str:
        patterns = [
            r'(?:youtube\.com/watch\?v=|youtu\.be/|youtube\.com/embed/)([a-zA-Z0-9_-]{11})'
        ]
        for pat in patterns:
            match = re.search(pat, url)
            if match:
                return match.group(1)
        return None
```
